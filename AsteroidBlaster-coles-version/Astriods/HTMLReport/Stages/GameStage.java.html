96       //Check if all asteroids are dead, if they are advance to the next stage
97       public void checkLevelUp(){ // handles level changing of player (goes up level if all asteroids on level are destroyed)
98           // if alien ship not destroyed, two alien ships added, need to destroy 
99           for(IGameObject item : this.gameObjects){
100              if(IAsteroid.class.isAssignableFrom(item.getClass()) && item.getIsAlive()) {
101                  return; // if asteroid was found, should not level up, thus we return from this function
102              }
103          }
104          this.level++; // if we reach this point, need to level up
105          for (int i = 0; i < this.level; i++) {
106              Asteroid asteroid = new Asteroid(AsteroidSize.Large); // create new asteroid based on current level
107              asteroid.onCreate(); 
108              this.gameObjects.add(asteroid);   // add to Game Object list
109          }
110          // each configured level to appear (currently set at 1)
111          if(this.level % this.settings.alienShipEveryLevel == 0){
112              this.gameObjects.add(new AlienShip(this.level * 3));
113          }
114      }

116      @Override
117      public void draw() { // gets called by GameEngine x amount of times per second
118          //The scene has not finished its initialization
119          if(root == null) { // to prevent NullPointerException
120              return;
121          }
122         
123          Platform.runLater(() -> {
124              //Add the queued bullets to the list, this is to avoid modfiying the list as you are looping
125              this.gameObjects.addAll(this.bulletQueue); // wait for loop to complete, first add all current bullets and the clear queue
126              
127              //clear the queue
128              this.bulletQueue =  new ArrayList<>();
129              root.getChildren().clear(); // clears all elements of form to be redrawn (show movement)
130              ArrayList<IGameObject> tempLst =  new ArrayList<>(); // used to add all objects still alive to scene
131              //Here we loop though all the objects to remove any dead objects --> determine if object is alive
132              for(IGameObject item : this.gameObjects){
133                  if(item.getIsAlive()){//if the item is still alive add it to the temp list
134                      tempLst.add(item);
135                  }else if(IAsteroid.class.isAssignableFrom(item.getClass())){ // check if object is asteroid that died, need to switch
136                      //recreate a smaller asteroid if Large or medium one dies
137                      switch(((IAsteroid)item).getSize()){ // convert to asteroid and add scoreLabel and two mediums, or small, depending on use case
138                          case Large: // logic varies depending on type of asteroid, what will need to be done, e.g large broken into two medium
139                                  this.score += 100;
140                                  tempLst.add(new Asteroid(AsteroidSize.Medium,item.getX(),item.getY()));
141                                  tempLst.add(new Asteroid(AsteroidSize.Medium,item.getX(),item.getY()));
142                              break;
143                          case Medium: // medium into small
144                                  this.score += 150;
145                                  tempLst.add(new Asteroid(AsteroidSize.Small,item.getX(),item.getY()));
146                                  tempLst.add(new Asteroid(AsteroidSize.Small,item.getX(),item.getY()));
147                              break;
148                          case Small: // small is destroyed and points added
149                                  this.score += 180;
150                              break;
151                      }
152                      // implement 1000 lives reached, granted additional life
153                      int scoreRounded = (int) Math.floor(this.score / 1000.0) * 1000;
154                      if (scoreRounded > this.livesGranted * 1000) {
155                          this.livesGranted++;
156                          this.ship.addLife();
157                      }
158                  }
159              }
160              
161              //Set the game objects to the temp list which only contains the alive objects
162              this.gameObjects = tempLst; // add all objects currently alive
163              //after all dead objects have been removed, check if there are any objects left on this level
164              this.checkLevelUp();
165              
166              //loop through all the game objects and call their update function
167              for(IGameObject item : this.gameObjects){
168                  item.update(); 
169                  
170                  //The polygon of the item to the root
171                  Polygon object = item.getPolygon(); // shape to be drawn
172                  root.getChildren().add(object); // add to form
173                
174                  //Logic should only execute on none player classes
175                  if(!IPlayer.class.isAssignableFrom(item.getClass())){ 
176                      //If the item is a bullet scan all the asteroids
177                      if(IBullet.class.isAssignableFrom(item.getClass())){
178                          //Check if this is a alien ship bullet, if it is check if the player ship has collided with it
179                         if(!((IBullet)item).getIsPlayerBullet()){ 
180                             //Check if the alien ship has shot the player ship
181                             if(ship.hasCollided(item)){
182                                 ship.subtractLife();
183                                 if(ship.getLives() == 0){
184                                      try {
185                                          this.gameOver();
186                                          return;//return out of this current loop not to show the stage.
187                                      } catch (Exception ex) {
188                                      }
189                                  }
190                             }
191                          }else{ // indicates Player's bullet
192                             //Check if any of the npc's (asteroid/alien ship) has collided with the bullets from the player ship
193                              for(IGameObject npc : this.gameObjects){
194                                  if(IAsteroid.class.isAssignableFrom(npc.getClass()) || IAlienShip.class.isAssignableFrom(npc.getClass())){
195                                      if(item.hasCollided(npc)){ 
196                                          npc.hasCollided(item);
197                                      }
198                                  }
199                              }
200                          }
201                      }
202                      // if not bullet and collision occurred with enemy object
203                      if(!IBullet.class.isAssignableFrom(item.getClass()) && ship.hasCollided(item)){
204                          ship.subtractLife();
205                          if(ship.getLives() == 0){
206                              try {
207                                  this.gameOver();
208                                  return;
209                              } catch (Exception ex) {
210                              }
211                          }
212                      }
213                  }
214              }
215              Text lives = new Text("LIVES: " + ship.getLives());
216              Text levelLabel = new Text("level: " + this.level);
217              levelLabel.setFill(Color.WHITE);
218              lives.setFill(Color.WHITE);
219              lives.setFont(new Font("Arial", 20));
220              levelLabel.setFont(new Font("Arial", 20));
221              lives.setX(scene.getWidth() - 100);
222              lives.setY(20);
223              Text scoreLabel = new Text("Score: " + this.score);
224              scoreLabel.setFill(Color.WHITE);
225              scoreLabel.setFont(new Font("Arial", 20));
226              scoreLabel.setX(20);
227              scoreLabel.setY(40);
228              levelLabel.setX(20);
229              levelLabel.setY(20);
230              root.getChildren().add(scoreLabel);
231              root.getChildren().add(lives);
232              root.getChildren().add(levelLabel);
233  
234              root.setStyle("-fx-background-color: black;");
235              Rectangle2D screenBounds = Screen.getPrimary().getVisualBounds();
236  
237              double centerX = (screenBounds.getWidth() - scene.getWidth()) / 2;
238              double centerY = (screenBounds.getHeight() - scene.getHeight()) / 2;
239  
240              stage.setX(centerX);
241              stage.setY(centerY);
242              stage.setScene(scene);
243              if(this.ship.getLives()> 0){
244                  stage.show();
245              }
246          });
247      }

249      @Override // only runs once
250      public void loadStage() { // called as soon as stage is set
251          this.score = 0; // initialise stage with all objects
252          this.level = 1;
253          this.livesGranted = 0;
254          this.gameObjects = new ArrayList<>();
255          // add new asteroid (only one on level one)
256          this.gameObjects.add(new Asteroid(AsteroidSize.Large)); 
257          // everything added to game object
258          this.settings = GameSettings.GetInstance(); 
259   
260          this.gameObjects.add(new AlienShip(this.level * 3)); // HP * 3 
261          // add ship to game object
262          this.ship = new Ship();
263          this.gameObjects.add(ship);
264          for(IGameObject item : this.gameObjects){
265                  item.onCreate();
266          }
267          // to run on main thread later
268          Platform.runLater(() -> {
269              this.stage = new Stage();
270              this.stage.initStyle(StageStyle.UNDECORATED);
271              this.root = new Pane();
272              this.scene = new Scene(root, settings.boardWidth,settings.boardHeight);
273              this.stage.setX(0);
274              this.stage.setY(0);
275              // key press events
276              this.scene.setOnKeyPressed((event) -> {
277                  // handles game pausing
278                  this.engine.handleKeyPress(event.getCode()); // check if code entered
279                  // handle player movement
280                  this.ship.handleKeyPress(event.getCode());
281               });
282              
283              this.scene.setOnKeyReleased((event) ->{
284                  // forces player to release key to enable shooting of bullet
285                  // prevent single string (make nice bullets) :p
286                  if(event.getCode() == KeyCode.SPACE){
287                      this.ship.shoot();
288                  }
289              } );
290            
291              // from GameEngine            
292              this.engine.addOnPauseEvent(this); 
293              this.draw(); // "draws" picture -- game loop of stage in GameStage
294          });
295      }

313      //Popup if the game is paused
314      @Override
315      public void onPause() {
316          Platform.runLater(() -> {
317              Stage popup = new Stage(); // generate 'pop-up' pause screen
318              popup.initStyle(StageStyle.UNDECORATED);
319              popup.initOwner(stage);
320              popup.initModality(Modality.APPLICATION_MODAL);
321  
322              VBox container = new VBox(10);
323              container.setAlignment(Pos.CENTER);
324              container.setPrefSize(250, 100);
325              container.setStyle("-fx-background-color: #f2f2f2; -fx-border-color: grey; -fx-border-width: 2;");
326              Label resumeLabel = new Label("Resume");
327              Label quitLabel = new Label("Quit");
328  
329              resumeLabel.setOnMouseClicked(e -> {
330                          popup.close(); 
331                          this.engine.resumeGame();
332                      });
333              quitLabel.setOnMouseClicked(e -> {
334                  popup.close();
335                  this.engine.resumeGame(); // back to main menu
336                  try {
337                      this.engine.setCurrentStage("MainStage");
338                  } catch (Exception ex) {
339                      ex.printStackTrace();
340                  }
341              });
342  
343              container.getChildren().addAll(resumeLabel, quitLabel);
344              Scene popupScene = new Scene(container);
345              popup.setScene(popupScene);
346              popup.show();
347      });
