1    package Logic;
2    
3    import Interfaces.IAlienShip;
4    import Interfaces.IBullet;
5    import Interfaces.IGameObject;
6    import Stages.GameStage;
7    import java.util.Random;
8    import javafx.scene.paint.Color;
9    import javafx.scene.shape.Polygon;
10   
11   //The alien ship class is created each round, the health of the ship is decided
12   //by the level x 3. The ship will fire at the players position.
13   public class AlienShip implements IGameObject, IAlienShip{
14       private int x;
15       private int y;
16       private final GameSettings settings ;
17       private final double speed = 5;
18       private int lives;
19       private final GameEngine engine;
20       private boolean isAlive = true;
21       private int currentXDestination;
22       private int currentYDestination;
23       private final Random rnd = new Random();
24       private long timeStamp =-1;
25       private final GameStage stage;
26       
27       //Default constructor
28       public AlienShip(int lives){
29           this.lives = lives;
30           this.settings = GameSettings.GetInstance();
31           this.engine = GameEngine.getEngine();
32           this.stage = ((GameStage)engine.getCurrentStage());
33       }
34       
35       @Override
36       public void onCreate() {
37       }
38   
39       @Override
40       public void update() {
41           
42           //If the timestamp value is still default (-1) no valid value has been assigned yet
43           if(timeStamp == -1){
44               this.timeStamp = System.currentTimeMillis();
45           }
46           
47           //Fire a bullet every .5 seconds
48           if(((System.currentTimeMillis() - this.timeStamp) >  500)){
49               this.shoot();
50               this.timeStamp = System.currentTimeMillis();
51           } 
52           
53           //If the ship is more or less 10 units within its destination set a new course for it
54           if((Math.abs(this.x - this.currentXDestination) < 10) && (Math.abs(this.y - this.currentYDestination) < 10)) {
55               setRandomDestination();
56           }
57           
58           //move the ship in the x direction
59           if(this.x != this.currentXDestination){
60               if(this.x < this.currentXDestination){
61                   this.x += speed;
62               }else{
63                   this.x -= speed;
64               }
65           }
66           //move the ship in the y direction
67           if(this.y != this.currentYDestination){
68               if(this.y < this.currentYDestination){
69                   this.y += speed;
70               }else {
71                   this.y -= speed;
72               }
73           }
74           //Return a new polygon with new destination
75           this.getPolygon();
76       }
77       
78       //Set a random destination for the ship based on the board
79       private void setRandomDestination(){
80           this.currentXDestination = rnd.nextInt(settings.boardHeight) - 60;        
81           this.currentYDestination = rnd.nextInt(settings.boardWidth) - 60;        
82       }
83       
84       //Draw the  ship
85       @Override
86       public Polygon getPolygon() {
87           double width = 60;
88           double height = 20;
89           Polygon polygon = new Polygon();
90           polygon.getPoints().addAll(new Double[]{
91               0.0, height / 2,
92               width / 3, height / 3,
93               width / 2, -height / 2,
94               2 * width / 3, height / 3,
95               width, height / 2,
96               2 * width / 3, 2 * height / 3,
97               width / 2, height / 2,
98               width / 3, 2 * height / 3
99           });
100          polygon.setTranslateX(x);
101          polygon.setTranslateY(y);
102          polygon.setStroke(Color.rgb(102, 255, 102));
103          polygon.setStrokeWidth(2.0);
104        
105          return polygon;
106      }
107      
108      //Return the ship Y pos
109      @Override
110      public int getY() {
111          return this.y;
112      }
113  
114      //Return the ship X pos
115      @Override
116      public int getX() {
117         return this.x;
118      }
119      
120      //Return the ships speed
121      @Override
122      public double getSpeed() {
123          return this.speed;
124      }
125  
126      //Check if the ship has collided with bullets from the player
127      @Override
128      public boolean hasCollided(IGameObject object) {
129          //check if the game object is a bullet. This is done by checking 
130          //if the game object implements the IBullet interface
131          if (object instanceof IBullet){
132              //cast the game object to a bullet 
133              IBullet bullet = (IBullet) object;
134              //Check if this bullet is a player bullet
135              if(bullet.getIsPlayerBullet()){ 
136                  //reduce the ship's lives
137                  this.lives--;
138                  //Update its Alive status based on its lives
139                  if(this.lives <= 0) {
140                      this.isAlive = false;
141                  }
142                  return true;
143              }
144          }
145          return false;
146      }
147  
148      @Override
149      public boolean getIsAlive() {
150         return this.isAlive;
151      }
152  
153      @Override
154      public void shoot() {
155          //Gets the player ships polygon
156          Polygon targetPolygon = ((GameStage)engine.getCurrentStage()).getShip().getPolygon();
157          
158          //create a new bullet which is not a player bullet
159          Bullet bullet = new Bullet((int)this.getPolygon().getBoundsInParent().getCenterX(), (int)this.getPolygon().getBoundsInParent().getCenterY(),false,this.speed);
160          
161          //Point the bullet in the direction of the player ship
162          double deltaX = targetPolygon.getBoundsInParent().getCenterX() - this.getPolygon().getBoundsInParent().getCenterX();
163          double deltaY = targetPolygon.getBoundsInParent().getCenterY() - this.getPolygon().getBoundsInParent().getCenterY();
164          double angle = Math.atan2(deltaY, deltaX);
165          bullet.getPolygon().setRotate(Math.toDegrees(angle));
166          
167          //Add it to the main stage bullet Queue
168          stage.addBulletToQueue(bullet);
169      }
170  }
