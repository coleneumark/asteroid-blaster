7    import java.io.IOException;
8    import java.util.ArrayList;
9    import java.util.List;
10   
11   //Class is used to handle anything score related
12   public class ScoreHandler {
13       //score text file
14       private final String filename = "high_score.txt";
15       
16       //function to add the score to the leaderboard
17       public void addScore(Score playerScore) throws IOException{
18           
19           //Check first if a file exists
20           this.createIfNotExists();
21           
22           //New top 10 list which we build
23           List<Score> scores = new ArrayList<>();
24           
25           //Current top 10 list
26           List<Score> top10 =  this.readScores();
27           
28           //If the score is empty add it
29           if(top10.isEmpty()){
30               scores.add(playerScore);
31               //store the score
32               this.writeScores(scores);
33           }else{
34               //Check to ensure the score is only added once
35               boolean added = false;
36               
37               for(Score item :top10){
38                   //If 10 records has been added write the file and return
39                   if(scores.size() == 10){
40                       this.writeScores(scores);
41                       return;
42                   }
43                   //if the score has not yet been added, and the score is more than 
44                   //the current score the player is point to add it above 
45                   //that score. If the count of the score has not reached
46                   //10 yet also add the old score after the player score
47                   if(!added && item.getScore() < playerScore.getScore()){
48   //                      System.out.println("Adding player score 1");
49                       added = true;
50                       scores.add(playerScore);
51                       if(scores.size() != 10){
52                           scores.add(item);
53                       }
54                   }else{
55                       scores.add(item);
56                   }
57               }
58               
59               // if the player did not beat any other players
60               // but there is not yet 10 records, also add the player's
61               //score to the scoreboard
62               if(!added && scores.size() !=  10){
63                   scores.add(playerScore);
64                   added = true;
65               }
66               if(added){
67                   this.writeScores(scores);
68               }
69           }
70       }
71       
72       //Checks if a file exists, if it does not it will create it
73       private void createIfNotExists() throws IOException{
74           File file = new File(filename);
75           file.createNewFile();
76       }
77       
78       //Write the new textfile containing the score
79       private void writeScores(List<Score> scores) throws IOException {
80           try (FileWriter writer = new FileWriter(filename)) {
81               writer.write(""); // Clear file content
82               for (Score score : scores) {
83                   writer.write(score.getName() + "|" + score.getScore() + "\n");
84               }
85           } // Clear file content
86       }
87       
88       //Read all the scores from the text file
89       public  List<Score> readScores() {
90           List<Score> scores = new ArrayList<>();
91           // self-disposing BufferReader
92           try (BufferedReader br = new BufferedReader(new FileReader(this.filename))) {
93               String line;
94               while ((line = br.readLine()) != null) {
95                   String[] parts = line.split("[|]");
96                   String name = parts[0];
97                   int score = Integer.parseInt(parts[1]);
98                   scores.add(new Score(score, name));
99               }
100          } catch (IOException e) {
101              e.printStackTrace();
102          }
103  
104          return scores;
105      }
