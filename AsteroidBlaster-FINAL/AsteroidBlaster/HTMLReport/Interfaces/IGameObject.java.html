13       public void update();
14

16       public Polygon getPolygon();
17       
18       //Returns the current y pos of the entity
19       public int getY();
20       
21       //Returns the current x pos of the entity
22       public int getX();
23       
24       //Gets the speed of the entity
25       public double getSpeed();
26

28       boolean hasCollided(IGameObject object);

31       public boolean getIsAlive();
