51       @Override
52       public void loadStage() { // called as soon as stage is set
53           Platform.runLater(() -> { // run on main thread as soon as it becomes available again for execution
54               this.stage = new Stage();
55               this.root = new Pane();
56               // Just some styling
57               this.scene = new Scene(root, settings.boardWidth,settings.boardHeight);
58               Label trustLabel = new Label("⬆: Increase Thrust");
59               Label shootLabel = new Label("Space: Shoot");
60               Label leftLabel = new Label("⬅: Rotate Left");
61               Label rightLabel = new Label("➡: Rotate Right");
62               Label decreaseTrustLabel = new Label("⬇: Decrease Thrust");
63               Label pause = new Label("Enter: Pause");
64               
65               trustLabel.setFont(new Font(20));
66               shootLabel.setFont(new Font(20));
67               leftLabel.setFont(new Font(20));
68               rightLabel.setFont(new Font(20));
69               decreaseTrustLabel.setFont(new Font(20));            
70               pause.setFont(new Font(20));
71   
72   
73               trustLabel.setTextFill(Color.WHITE);
74               shootLabel.setTextFill(Color.WHITE);
75               leftLabel.setTextFill(Color.WHITE);
76               rightLabel.setTextFill(Color.WHITE);
77               decreaseTrustLabel.setTextFill(Color.WHITE);
78               pause.setTextFill(Color.WHITE);
79               
80               trustLabel.setAlignment(Pos.CENTER);
81               shootLabel.setAlignment(Pos.CENTER);
82               leftLabel.setAlignment(Pos.CENTER);
83               rightLabel.setAlignment(Pos.CENTER);
84               decreaseTrustLabel.setAlignment(Pos.CENTER);
85               pause.setAlignment(Pos.CENTER);
86               
87               root.getChildren().addAll(trustLabel, shootLabel, leftLabel, rightLabel, decreaseTrustLabel,pause);
88   
89               trustLabel.layoutXProperty().bind(root.widthProperty().divide(2).subtract(trustLabel.widthProperty().divide(2)));
90               shootLabel.layoutXProperty().bind(root.widthProperty().divide(2).subtract(shootLabel.widthProperty().divide(2)));
91               leftLabel.layoutXProperty().bind(root.widthProperty().divide(2).subtract(leftLabel.widthProperty().divide(2)));
92               rightLabel.layoutXProperty().bind(root.widthProperty().divide(2).subtract(rightLabel.widthProperty().divide(2)));
93               decreaseTrustLabel.layoutXProperty().bind(root.widthProperty().divide(2).subtract(decreaseTrustLabel.widthProperty().divide(2)));
94               pause.layoutXProperty().bind(root.widthProperty().divide(2).subtract(pause.widthProperty().divide(2)));
95               
96               trustLabel.layoutYProperty().bind(root.heightProperty().divide(2).add(trustLabel.heightProperty().multiply(3.0)));
97               decreaseTrustLabel.layoutYProperty().bind(root.heightProperty().divide(2).add(decreaseTrustLabel.heightProperty().multiply(4)));
98               leftLabel.layoutYProperty().bind(root.heightProperty().divide(2).add(leftLabel.heightProperty().multiply(5)));
99               rightLabel.layoutYProperty().bind(root.heightProperty().divide(2).add(rightLabel.heightProperty().multiply(6)));
100              shootLabel.layoutYProperty().bind(root.heightProperty().divide(2).add(shootLabel.heightProperty().multiply(7)));
101              pause.layoutYProperty().bind(root.heightProperty().divide(2).add(pause.heightProperty().multiply(8)));
102  
103              
104              Label playLabel = new Label("Play");
105              Label highscoreLabel = new Label("Highscore");
106              Label quitLabel = new Label("Quit");
107              root.setStyle("-fx-background-color: black;");
108              playLabel.setTextFill(Color.WHITE);
109              quitLabel.setTextFill(Color.WHITE);
110              highscoreLabel.setTextFill(Color.WHITE);
111              
112              quitLabel.setOnMouseClicked(event -> {
113                  // code to be executed when label is clicked
114                  System.exit(0);
115               });
116              
117              highscoreLabel.setOnMouseClicked(event -> {
118                  try {
119                      // code to be executed when label is clicked
120                      engine.setCurrentStage("HighScoreStage");
121                  } catch (Exception ex) {
122                      ex.printStackTrace();
123                  }
124               });
125              
126              playLabel.setOnMouseClicked(event -> {
127                  try {
128                      // code to be executed when label is clicked
129                      engine.setCurrentStage("GameStage");
130                  } catch (Exception ex) {
131                      ex.printStackTrace();
132                  }
133               });
134  
135              playLabel.setFont(new Font(36));
136              highscoreLabel.setFont(new Font(36));
137              quitLabel.setFont(new Font(36));
138              
139              playLabel.setAlignment(Pos.CENTER);
140              highscoreLabel.setAlignment(Pos.CENTER);
141              quitLabel.setAlignment(Pos.CENTER);
142  
143              this.stage.initStyle(StageStyle.UNDECORATED);
144              // Add the labels to the root pane and position them in the center
145              root.getChildren().addAll(playLabel, highscoreLabel, quitLabel);
146              root.setCenterShape(true);
147  
148              playLabel.layoutXProperty().bind(root.widthProperty().divide(2).subtract(playLabel.widthProperty().divide(2)));
149              highscoreLabel.layoutXProperty().bind(root.widthProperty().divide(2).subtract(highscoreLabel.widthProperty().divide(2)));
150              quitLabel.layoutXProperty().bind(root.widthProperty().divide(2).subtract(quitLabel.widthProperty().divide(2)));        
151  
152              playLabel.layoutYProperty().bind(root.heightProperty().divide(2).subtract(playLabel.heightProperty().multiply(1.5)));
153              highscoreLabel.layoutYProperty().bind(root.heightProperty().divide(2).subtract(highscoreLabel.heightProperty().multiply(0.5)));
154              quitLabel.layoutYProperty().bind(root.heightProperty().divide(2).add(quitLabel.heightProperty().multiply(0.5)));
155  
156              Rectangle2D screenBounds = Screen.getPrimary().getVisualBounds();
157              double centerX = (screenBounds.getWidth() - scene.getWidth()) / 2;
158              double centerY = (screenBounds.getHeight() - scene.getHeight()) / 2;
159  
160              stage.setX(centerX);
161              stage.setY(centerY);
162              stage.setScene(scene);
163              this.stage.show();
164          });
165      }
